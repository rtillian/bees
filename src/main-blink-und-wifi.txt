#include <Arduino.h>
#include <WiFi.h>

const char* ssid     = "TP-Link_8CFC";
const char* password = "83648168";
void setup() {
  pinMode(26, OUTPUT);
  pinMode(2, OUTPUT); 
  Serial.begin(115200);
  
  
  Serial.println("Anfang");
  delay(10000);
  Serial.println("Verbinde mit WLAN...");
  WiFi.begin(ssid, password);

  // Warten, bis Verbindung steht
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nVerbunden!");
  Serial.print("IP-Adresse: ");
  Serial.println(WiFi.localIP());
  
}

void loop() {
  digitalWrite(26, HIGH);  // LED an
  delay(1000);
  Serial.println("eins");

  digitalWrite(26, LOW);   // LED aus
  delay(1000);
  Serial.println("zwei");
  digitalWrite(2, HIGH);
  delay(500);
  digitalWrite(2, LOW);
  delay (200);
}
